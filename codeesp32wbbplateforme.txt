// ===================== HELPERS =====================
static inline float clampf(float x, float a, float b) {
  if (x < a) return a;
  if (x > b) return b;
  return x;
}
static inline int iClamp(int v, int lo, int hi) {
  if (v < lo) return lo;
  if (v > hi) return hi;
  return v;
}
static inline int iMax(int a, int b) { return (a > b) ? a : b; }

// ===================== VL53 =====================
VL53L0X vl53;
bool vl53Ok = false;

static const int POS_N = 1;
int posBuf[POS_N] = {0};
int posIdx = 0;
bool posFilled = false;

int readPosRaw() {
  if (!vl53Ok) return -1;
  uint16_t d = vl53.readRangeContinuousMillimeters();
  if (vl53.timeoutOccurred()) return -1;
  return (int)d;
}
int readPosFiltered() {
  int p = readPosRaw();
  if (p < 0) return -1;

  posBuf[posIdx] = p;
  posIdx = (posIdx + 1) % POS_N;
  if (posIdx == 0) posFilled = true;

  int count = posFilled ? POS_N : posIdx;
  long sum = 0;
  for (int i = 0; i < count; i++) sum += posBuf[i];
  return (int)(sum / count);
}

// ===================== MOTOR =====================
void motorStop() {
  ledcWrite(PIN_RPWM, 0);
  ledcWrite(PIN_LPWM, 0);
  digitalWrite(PIN_REN, LOW);
  digitalWrite(PIN_LEN, LOW);
}

void motorSetSigned(int pwmSigned) {
  if (invertDir) pwmSigned = -pwmSigned;

  int pwm = abs(pwmSigned);
  if (pwm <= 0) { motorStop(); return; }
  if (pwm > 255) pwm = 255;

  digitalWrite(PIN_REN, HIGH);
  digitalWrite(PIN_LEN, HIGH);

  if (pwmSigned > 0) { // FRONT
    ledcWrite(PIN_RPWM, 0);
    ledcWrite(PIN_LPWM, pwm);
  } else {            // BACK
    ledcWrite(PIN_LPWM, 0);
    ledcWrite(PIN_RPWM, pwm);
  }
}

void motorTowardFront(int pwm) { motorSetSigned(+pwm); }
void motorTowardBack (int pwm) { motorSetSigned(-pwm); }

// ===================== SWITCH DEBOUNCE =====================
uint32_t swF_lastChange = 0, swB_lastChange = 0;
int swF_stable = 0, swB_stable = 0;
int swF_rawPrev = 0, swB_rawPrev = 0;
static const uint32_t SW_STABLE_MS = 20;

void updateSwitchDebounce() {
  int rf = digitalRead(PIN_SW_FRONT);
  int rb = digitalRead(PIN_SW_BACK);
  uint32_t now = millis();

  if (rf != swF_rawPrev) { swF_rawPrev = rf; swF_lastChange = now; }
  if (rb != swB_rawPrev) { swB_rawPrev = rb; swB_lastChange = now; }

  if (now - swF_lastChange >= SW_STABLE_MS) swF_stable = swF_rawPrev;
  if (now - swB_lastChange >= SW_STABLE_MS) swB_stable = swB_rawPrev;
}

inline bool swFrontPressed() { return swF_stable == HIGH; }
inline bool swBackPressed()  { return swB_stable == HIGH; }

// ===================== SAFE RANGE =====================
void computeSafeRange() {
  if (posMinMm < 0 || posMaxMm < 0) { posMinSafe = posMaxSafe = -1; return; }
  posMinSafe = posMinMm + endMarginMm;
  posMaxSafe = posMaxMm - endMarginMm;
  if (posMinSafe > posMaxSafe) { posMinSafe = posMinMm; posMaxSafe = posMaxMm; }

  Serial.print("SAFE_RANGE minSafe="); Serial.print(posMinSafe);
  Serial.print(" maxSafe="); Serial.println(posMaxSafe);
}

// ===================== NVS save/load =====================
void saveCalibration() {
  prefs.begin(NVS_NS, false);
  prefs.putUInt("magic", CAL_MAGIC);
  prefs.putInt("min", posMinMm);
  prefs.putInt("max", posMaxMm);
  prefs.putInt("center", centerMm);
  prefs.putInt("minS", posMinSafe);
  prefs.putInt("maxS", posMaxSafe);
  prefs.putBool("fim", frontIsMin);
  prefs.putInt("margin", endMarginMm);
  prefs.putBool("inv", invertDir);
  prefs.putBool("copinv", copInvert);
  prefs.putInt("slowZ", slowZoneMm);
  prefs.putInt("slowP", slowPwmMax);
  prefs.putInt("relp", releasePwm);
  prefs.putUInt("relt", releaseToMs);
  prefs.putInt("hstp", hardStopMm);
  prefs.putInt("pminm", pwmMinMove);
  prefs.putInt("kpwm", kickPwm);
  prefs.putInt("kms", kickMs);
  prefs.end();
  Serial.println("CAL: saved");
}

bool loadCalibration() {
  prefs.begin(NVS_NS, true);
  uint32_t m = prefs.getUInt("magic", 0);
  if (m != CAL_MAGIC) { prefs.end(); return false; }

  posMinMm = prefs.getInt("min", -1);
  posMaxMm = prefs.getInt("max", -1);
  centerMm = prefs.getInt("center", -1);
  posMinSafe = prefs.getInt("minS", -1);
  posMaxSafe = prefs.getInt("maxS", -1);
  frontIsMin = prefs.getBool("fim", true);
  endMarginMm = prefs.getInt("margin", endMarginMm);
  invertDir = prefs.getBool("inv", invertDir);
  copInvert = prefs.getBool("copinv", copInvert);
  slowZoneMm = prefs.getInt("slowZ", slowZoneMm);
  slowPwmMax = prefs.getInt("slowP", slowPwmMax);
  releasePwm = prefs.getInt("relp", releasePwm);
  releaseToMs = prefs.getUInt("relt", releaseToMs);
  hardStopMm = prefs.getInt("hstp", hardStopMm);
  pwmMinMove = prefs.getInt("pminm", pwmMinMove);
  kickPwm = prefs.getInt("kpwm", kickPwm);
  kickMs = prefs.getInt("kms", kickMs);
  prefs.end();

  if (posMinMm < 0 || posMaxMm < 0) return false;
  if (centerMm < 0) centerMm = (posMinMm + posMaxMm) / 2;
  if (posMinSafe < 0 || posMaxSafe < 0) computeSafeRange();

  homed = true;
  Serial.println("CAL: loaded (homed=1)");
  return true;
}

void clearCalibration() {
  prefs.begin(NVS_NS, false);
  prefs.clear();
  prefs.end();
  Serial.println("CAL: cleared");
  homed = false;
}

// ===================== RELEASE =====================
bool waitReleasedStable(bool (*pressedFn)(), uint32_t stableMs, uint32_t timeoutMs) {
  uint32_t t0 = millis();
  uint32_t stableStart = 0;
  while (millis() - t0 < timeoutMs) {
    updateSwitchDebounce();
    if (!pressedFn()) {
      if (stableStart == 0) stableStart = millis();
      if (millis() - stableStart >= stableMs) return true;
    } else {
      stableStart = 0;
    }
    delay(2);
  }
  return false;
}

bool releaseFrontHard() {
  updateSwitchDebounce();
  if (!swFrontPressed()) return true;

  Serial.println("REL: FRONT pressed -> backing off...");
  uint32_t t0 = millis();
  while (swFrontPressed() && (millis() - t0 < releaseToMs)) {
    motorTowardBack(releasePwm);
    delay(5);
    updateSwitchDebounce();
  }
  motorStop(); delay(80);

  bool ok = waitReleasedStable(swFrontPressed, 25, 600);
  if (!ok) { Serial.println("REL ERROR: FRONT still pressed"); return false; }
  Serial.println("REL OK: FRONT released");
  return true;
}

bool releaseBackHard() {
  updateSwitchDebounce();
  if (!swBackPressed()) return true;

  Serial.println("REL: BACK pressed -> moving off...");
  uint32_t t0 = millis();
  while (swBackPressed() && (millis() - t0 < releaseToMs)) {
    motorTowardFront(releasePwm);
    delay(5);
    updateSwitchDebounce();
  }
  motorStop(); delay(80);

  bool ok = waitReleasedStable(swBackPressed, 25, 600);
  if (!ok) { Serial.println("REL ERROR: BACK still pressed"); return false; }
  Serial.println("REL OK: BACK released");
  return true;
}

bool releaseAny() {
  bool okF = releaseFrontHard();
  bool okB = releaseBackHard();
  return okF && okB;
}

// ===================== CENTERING =====================
bool wantTowardFront_fromTargetPos(int target, int pos) {
  return frontIsMin ? (target < pos) : (target > pos);
}

bool goToTargetBlocking(int tgtMm, uint32_t timeoutMs) {
  if (!armed || !homed) return false;
  Serial.print("CENTER: target="); Serial.println(tgtMm);

  uint32_t t0 = millis();
  while (millis() - t0 < timeoutMs) {
    updateSwitchDebounce();
    if (swFrontPressed() || swBackPressed()) {
      if (!releaseAny()) return false;
      delay(100);
    }

    int p = readPosFiltered();
    if (p < 0) { motorStop(); return false; }
    posMm = p;

    int err = tgtMm - posMm;
    if (abs(err) < 4) { motorStop(); Serial.println("CENTER: reached"); return true; }

    bool wantFront = wantTowardFront_fromTargetPos(tgtMm, posMm);
    if (wantFront) motorTowardFront(centerPwm);
    else           motorTowardBack(centerPwm);

    delay(10);
  }
  motorStop();
  Serial.println("CENTER: timeout");
  return false;
}

// ===================== HOMING =====================
bool doHoming() {
  Serial.println("=== HOMING START ===");
  if (!armed) { Serial.println("Homing ERROR: not armed"); return false; }

  autoMode = false;
  homed = false;
  cmdNow = 0;
  kickUntilMs = 0;
  zeroStartMs = millis();

  swF_rawPrev = digitalRead(PIN_SW_FRONT);
  swB_rawPrev = digitalRead(PIN_SW_BACK);
  swF_stable = swF_rawPrev;
  swB_stable = swB_rawPrev;
  swF_lastChange = swB_lastChange = millis();

  Serial.println("Homing -> BACK ...");
  uint32_t t0 = millis();
  while (true) {
    updateSwitchDebounce();
    if (swBackPressed()) break;
    motorTowardBack(homingPwm);
    if (millis() - t0 > 20000) { motorStop(); Serial.println("Homing ERROR: timeout BACK"); return false; }
    delay(5);
  }
  motorStop(); delay(200);
  int backAt = readPosRaw();
  Serial.print("BACK reached. backAtSwitchMm="); Serial.println(backAt);
  if (!releaseBackHard()) return false;

  Serial.println("Homing -> FRONT ...");
  t0 = millis();
  while (true) {
    updateSwitchDebounce();
    if (swFrontPressed()) break;
    motorTowardFront(homingPwm);
    if (millis() - t0 > 20000) { motorStop(); Serial.println("Homing ERROR: timeout FRONT"); return false; }
    delay(5);
  }
  motorStop(); delay(200);
  int frontAt = readPosRaw();
  Serial.print("FRONT reached. frontAtSwitchMm="); Serial.println(frontAt);

  if (frontAt < 0 || backAt < 0) { Serial.println("Homing ERROR: VL53 invalid"); return false; }

  frontIsMin = (frontAt < backAt);
  posMinMm = min(frontAt, backAt);
  posMaxMm = max(frontAt, backAt);
  centerMm = (posMinMm + posMaxMm) / 2;

  Serial.print("frontIsMin="); Serial.println(frontIsMin ? 1 : 0);
  Serial.print("CENTER_MM=");  Serial.println(centerMm);
  Serial.print("RANGE_MM=");   Serial.println(posMaxMm - posMinMm);

  if (!releaseFrontHard()) return false;

  computeSafeRange();
  saveCalibration();
  homed = true;

  Serial.println("Homing -> GO CENTER ...");
  bool okCenter = goToTargetBlocking(centerMm, 12000);
  Serial.print("CENTER_OK="); Serial.println(okCenter ? 1 : 0);

  Serial.println("=== HOMING DONE ===");
  return okCenter;
}

// ===================== COP -> CMD =====================
int computeCmdFromCop(float cop) {
  float c = clampf(cop, -copClamp, +copClamp);
  if (fabs(c) < copDead) return 0;

  float norm = c / copClamp;
  if (copInvert) norm = -norm;

  float expo = 0.25f;                 // 0.5..0.8 recommandé
  float a = powf(fabs(norm), expo);  // expo < 1 => plus agressif près de 0

  int pwm = (int)lroundf(vPwmMin + a * (vPwmMax - vPwmMin));
  pwm = iClamp(pwm, 0, 255);

  int signedPwm = (norm > 0) ? +pwm : -pwm;

  if (signedPwm != 0 && abs(signedPwm) < pwmMinMove) {
    signedPwm = (signedPwm > 0) ? +pwmMinMove : -pwmMinMove;
  }
  return signedPwm;
}

// Rampe continue
int rampCmd(int cmdTarget) {
  int delta = cmdTarget - cmdNow;
  if (delta >  slewPerStep) delta =  slewPerStep;
  if (delta < -slewPerStep) delta = -slewPerStep;
  cmdNow += delta;

  if (abs(cmdNow) < 2 && abs(cmdTarget) < 2) cmdNow = 0;
  return cmdNow;
}

int applySlowZoneLimit(int cmdTarget, int pos) {
  if (posMinSafe < 0 || posMaxSafe < 0) return cmdTarget;
  if (cmdTarget == 0) return 0;

  bool cmdFront = (cmdTarget > 0);
  bool headingToMin = frontIsMin ? cmdFront : !cmdFront;

  int dist = headingToMin ? (pos - posMinSafe) : (posMaxSafe - pos);
  if (dist < 0) dist = 0;

  if (dist <= slowZoneMm) {
    int sign = (cmdTarget > 0) ? +1 : -1;
    int mag = abs(cmdTarget);
    if (mag > slowPwmMax) mag = slowPwmMax;
    return sign * mag;
  }
  return cmdTarget;
}

bool shouldHardStop(int cmdTarget, int pos) {
  if (posMinSafe < 0 || posMaxSafe < 0) return false;
  if (cmdTarget == 0) return false;

  bool cmdFront = (cmdTarget > 0);
  bool headingToMin = frontIsMin ? cmdFront : !cmdFront;

  int dist = headingToMin ? (pos - posMinSafe) : (posMaxSafe - pos);
  return dist <= hardStopMm;
}

// Kick seulement après un vrai arrêt
void maybeKickStart(int cmdTarget) {
  if (kickMs <= 0) return;
  if (cmdNow != 0) return;
  if (cmdTarget == 0) return;
  if (zeroStartMs == 0) return;
  if (millis() - zeroStartMs < kickMinStopMs) return;

  kickUntilMs = millis() + (uint32_t)kickMs;
  int s = (cmdTarget > 0) ? +1 : -1;
  motorSetSigned(s * kickPwm);
}

void autoStep() {
  if (!armed || !homed || !autoMode) return;

  updateSwitchDebounce();
  uint32_t now = millis();

  if (now - lastCopMs > copWatchdogMs) {
    motorSetSigned(rampCmd(0));
    if (cmdNow == 0 && zeroStartMs == 0) zeroStartMs = now;
    if (cmdNow != 0) zeroStartMs = 0;
    return;
  }

  if (kickUntilMs != 0 && now < kickUntilMs) return;
  if (kickUntilMs != 0 && now >= kickUntilMs) kickUntilMs = 0;

  if (swFrontPressed()) {
    motorStop(); cmdNow = 0; kickUntilMs = 0;
    zeroStartMs = now;
    blockTowardFrontUntilMs = now + switchCooldownMs;
    releaseFrontHard();
    return;
  }
  if (swBackPressed()) {
    motorStop(); cmdNow = 0; kickUntilMs = 0;
    zeroStartMs = now;
    blockTowardBackUntilMs = now + switchCooldownMs;
    releaseBackHard();
    return;
  }

  int p = readPosFiltered();
  if (p < 0) { motorStop(); cmdNow = 0; kickUntilMs = 0; zeroStartMs = now; return; }
  posMm = p;

  int cmdTarget = computeCmdFromCop(copY);

  if (cmdTarget > 0 && now < blockTowardFrontUntilMs) cmdTarget = 0;
  if (cmdTarget < 0 && now < blockTowardBackUntilMs)  cmdTarget = 0;

  if (shouldHardStop(cmdTarget, posMm)) {
    motorStop(); cmdNow = 0; kickUntilMs = 0;
    cmdTarget = 0;
  } else {
    cmdTarget = applySlowZoneLimit(cmdTarget, posMm);
  }

  if (cmdTarget == 0 && cmdNow == 0) {
    if (zeroStartMs == 0) zeroStartMs = now;
  } else {
    zeroStartMs = 0;
  }

  maybeKickStart(cmdTarget);
  if (kickUntilMs != 0) return;

  int out = rampCmd(cmdTarget);
  motorSetSigned(out);

  static uint32_t lastDbg = 0;
  if (dbg && millis() - lastDbg > 200) {
    lastDbg = millis();
    Serial.print("[AUTO_V] cop="); Serial.print(copY, 3);
    Serial.print(" tgt="); Serial.print(cmdTarget);
    Serial.print(" out="); Serial.print(out);
    Serial.print(" pos="); Serial.print(posMm);
    Serial.print(" swF="); Serial.print(swFrontPressed()?1:0);
    Serial.print(" swB="); Serial.println(swBackPressed()?1:0);
  }
}

bool looksLikeGarbage(const String& s) {
  if (s == "STOP" || s == "STAT" || s == "REL" || s == "HOME" || s == "HOME:1" ||
      s == "CENTER" || s == "CENTER:1") return false;

  if (s.startsWith("ARM:") || s.startsWith("AUTO:") || s.startsWith("COP:") || s.startsWith("COP:Y:")) return false;
  if (s.startsWith("CAL:") || s.startsWith("COPINV:") || s.startsWith("DBG:COP:") || s.startsWith("SET:")) return false;

  bool hasColon = (s.indexOf(':') >= 0);
  bool hasAlpha = false;
  for (int i=0;i<s.length();i++) if (isAlpha((unsigned char)s[i])) { hasAlpha=true; break; }
  if (!hasColon && !hasAlpha) return true;
  return false;
}

void handleLine(const String& sRaw) {
  String s = sRaw; s.trim();
  if (!s.length()) return;
  if (looksLikeGarbage(s)) return;

  if (s.startsWith("ARM:")) {
    armed = (s.substring(4).toInt() != 0);
    if (!armed) { motorStop(); autoMode=false; cmdNow=0; kickUntilMs=0; zeroStartMs=millis(); }
    Serial.print("ARMED="); Serial.println(armed?1:0);
    return;
  }

  if (s == "HOME" || s == "HOME:1") {
    bool ok = doHoming();
    Serial.print("HOME_OK="); Serial.println(ok?1:0);
    return;
  }

  // ✅ NOUVEAU : recentrage au centre (via VL53 + centerMm)
  if (s == "CENTER" || s == "CENTER:1") {
    if (!armed || !homed) {
      Serial.println("CENTER ERROR: need ARM=1 and homed=1");
      Serial.print("CENTER_OK="); Serial.println(0);
      return;
    }
    autoMode = false;        // on coupe l'auto
    motorStop();
    cmdNow = 0;
    kickUntilMs = 0;
    zeroStartMs = millis();

    bool ok = goToTargetBlocking(centerMm, 12000);
    Serial.print("CENTER_OK="); Serial.println(ok ? 1 : 0);
    return;
  }

  if (s.startsWith("AUTO:")) {
    autoMode = (s.substring(5).toInt() != 0);
    if (!autoMode) { motorStop(); cmdNow=0; kickUntilMs=0; zeroStartMs=millis(); }
    else { cmdNow=0; kickUntilMs=0; zeroStartMs=millis(); }
    Serial.print("AUTO="); Serial.println(autoMode?1:0);
    return;
  }

  if (s.startsWith("COP:Y:")) {
    String v = s.substring(6);
    copY = v.toFloat();
    lastCopMs = millis();
    if (dbgCop) { Serial.print("RX_COP:'"); Serial.print(v); Serial.println("'"); }
    return;
  }
  if (s.startsWith("COP:")) {
    String v = s.substring(4);
    copY = v.toFloat();
    lastCopMs = millis();
    if (dbgCop) { Serial.print("RX_COP:'"); Serial.print(v); Serial.println("'"); }
    return;
  }

  if (s == "REL") { bool ok = releaseAny(); Serial.print("REL_ANY="); Serial.println(ok?1:0); return; }
  if (s == "STAT") { printState(); return; }
  if (s == "PARAMS?" || s == "PARAMS") {
    Serial.print("PARAMS ");
    Serial.print("SLEW="); Serial.print(slewPerStep);
    Serial.print(" PMIN="); Serial.print(pwmMinMove);
    Serial.print(" KICKPWM="); Serial.print(kickPwm);
    Serial.print(" KICKMS="); Serial.print(kickMs);
    Serial.print(" ENDMARGIN="); Serial.print(endMarginMm);
    Serial.print(" SLOWZONE="); Serial.print(slowZoneMm);
    Serial.print(" SLOWPWM="); Serial.print(slowPwmMax);
    Serial.print(" HARDSTOP="); Serial.print(hardStopMm);
    Serial.print(" COPINV="); Serial.print(copInvert ? 1 : 0);
    Serial.println();
    return;
  }

  if (s == "CAL:CLR") { clearCalibration(); return; }

  if (s.startsWith("COPINV:")) {
    copInvert = (s.substring(7).toInt() != 0);
    saveCalibration();
    Serial.print("COPINV="); Serial.println(copInvert?1:0);
    return;
  }

  if (s.startsWith("DBG:COP:")) {
    dbgCop = (s.substring(8).toInt() != 0);
    Serial.print("DBG_COP="); Serial.println(dbgCop?1:0);
    return;
  }
  
  if (s == "PROFILE:SOT") {
    // Preset SOT (tes valeurs actuelles)
    slewPerStep  = 60;
    pwmMinMove   = 30;
    kickPwm      = 140;
    kickMs       = 50;
    endMarginMm  = 12;
    slowZoneMm   = 8;
    slowPwmMax   = 90;
    hardStopMm   = 2;

    computeSafeRange();
    saveCalibration();

    Serial.println("PROFILE=SOT OK");
    return;
  }

  if (s.startsWith("SET:SLEW:")) {
    slewPerStep = iClamp((int)s.substring(9).toInt(), 1, 50);
    Serial.print("SLEW="); Serial.println(slewPerStep);
    return;
  }
  if (s.startsWith("SET:ENDMARGIN:")) {
    endMarginMm = iMax(0, (int)s.substring(14).toInt());
    computeSafeRange(); saveCalibration();
    Serial.print("ENDMARGIN="); Serial.println(endMarginMm);
    return;
  }
  if (s.startsWith("SET:SLOWZONE:")) {
    // "SET:SLOWZONE:" = 13 caractères -> index 13
    slowZoneMm = iMax(0, (int)s.substring(13).toInt());
    saveCalibration();
    Serial.print("SLOWZONE="); Serial.println(slowZoneMm);
    return;
  }
  if (s.startsWith("SET:SLOWPWM:")) {
    slowPwmMax = iClamp((int)s.substring(12).toInt(), 0, 255);
    saveCalibration();
    Serial.print("SLOWPWM="); Serial.println(slowPwmMax);
    return;
  }
  if (s.startsWith("SET:REL:PWM:")) {
    releasePwm = iClamp((int)s.substring(12).toInt(), 0, 255);
    saveCalibration();
    Serial.print("REL_PWM="); Serial.println(releasePwm);
    return;
  }
  if (s.startsWith("SET:REL:TO:")) {
    releaseToMs = (uint32_t)iMax(100, (int)s.substring(11).toInt());
    saveCalibration();
    Serial.print("REL_TO_MS="); Serial.println(releaseToMs);
    return;
  }
  if (s.startsWith("SET:HARDSTOP:")) {
    hardStopMm = iClamp((int)s.substring(13).toInt(), 0, 50);
    saveCalibration();
    Serial.print("HARDSTOP_MM="); Serial.println(hardStopMm);
    return;
  }
  if (s.startsWith("SET:PMIN:")) {
    pwmMinMove = iClamp((int)s.substring(9).toInt(), 0, 255);
    saveCalibration();
    Serial.print("PMIN="); Serial.println(pwmMinMove);
    return;
  }
  if (s.startsWith("SET:KICKPWM:")) {
    kickPwm = iClamp((int)s.substring(12).toInt(), 0, 255);
    saveCalibration();
    Serial.print("KICKPWM="); Serial.println(kickPwm);
    return;
  }
  if (s.startsWith("SET:KICKMS:")) {
    kickMs = iClamp((int)s.substring(11).toInt(), 0, 500);
    saveCalibration();
    Serial.print("KICKMS="); Serial.println(kickMs);
    return;
  }

  if (s == "STOP") {
    motorStop();
    armed=false; autoMode=false; cmdNow=0; kickUntilMs=0; zeroStartMs=millis();
    Serial.println("STOP OK");
    return;
  }

  Serial.print("Unknown cmd: "); Serial.println(s);
}

void readLines(Stream& ser, String& buf) {
  while (ser.available()) {
    char c = (char)ser.read();
    if (c=='\r') continue;
    if (c=='\n') {
      String s = buf; buf="";
      handleLine(s);
    } else {
      if (buf.length() < 200) buf += c;
    }
  }
}

// ===================== Setup / Loop =====================
void setup() {
  Serial.begin(115200);
  delay(1200);

  pinMode(PIN_SW_BACK,  INPUT);
  pinMode(PIN_SW_FRONT, INPUT);

  pinMode(PIN_REN, OUTPUT);
  pinMode(PIN_LEN, OUTPUT);

  ledcAttach(PIN_RPWM, PWM_FREQ, PWM_RES);
  ledcAttach(PIN_LPWM, PWM_FREQ, PWM_RES);

  motorStop();

  Wire.begin(21,22);
  vl53.setTimeout(200);
  if (!vl53.init()) {
    Serial.println("VL53 init FAILED");
    vl53Ok=false;
  } else {
    Serial.println("VL53 init OK");
    vl53Ok=true;
    vl53.startContinuous(15);
  }

  Serial2.begin(115200, SERIAL_8N1, UART_RX, UART_TX);

  swF_rawPrev = digitalRead(PIN_SW_FRONT);
  swB_rawPrev = digitalRead(PIN_SW_BACK);
  swF_stable = swF_rawPrev;
  swB_stable = swB_rawPrev;
  swF_lastChange = swB_lastChange = millis();

  zeroStartMs = millis();

  Serial.println("BOOT...");
  if (!loadCalibration()) Serial.println("CAL: not found (run HOME)");
  Serial.println("READY");
  Serial.println("Cmds: ARM:1 HOME CENTER AUTO:1 COP:Y:x STOP STAT REL CAL:CLR COPINV:0/1 DBG:COP:1 SET:PMIN:x SET:KICKPWM:x SET:KICKMS:x");
}

void loop() {
  readLines(Serial,  bufUSB);
  readLines(Serial2, bufUART);

  static uint32_t lastStep = 0;
  uint32_t now = millis();

  // Switch debounce souvent (pas cher)
  updateSwitchDebounce();

  // lecture position à 5ms (200 Hz max) — le VL53 répondra au rythme interne
  static uint32_t lastPos = 0;
  if (now - lastPos >= 5) {
    lastPos = now;
    int p = readPosFiltered();
    if (p >= 0) posMm = p;
  }

  // autoStep très rapide (2ms)
  if (now - lastStep >= 2) {
    lastStep = now;
    autoStep();
  }
}
